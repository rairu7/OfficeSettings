' 一覧で選択したメールの差出人アドレスを自動的にクリップボードに入れる
' ショートカット：Ctrl＋Shift＋C
' 2025/10/15
Sub CopySenderAddress()
    Dim mail As Outlook.MailItem
    If TypeOf Application.ActiveExplorer.Selection(1) Is Outlook.MailItem Then
        Set mail = Application.ActiveExplorer.Selection(1)
        Dim DataObj As Object
        Set DataObj = CreateObject("MSForms.DataObject")
        DataObj.SetText mail.SenderEmailAddress
        DataObj.PutInClipboard
        MsgBox "差出人アドレスをコピーしました:" & vbCrLf & mail.SenderEmailAddress, vbInformation
    End If
End Sub
