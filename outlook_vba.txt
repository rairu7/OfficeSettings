Private Const GMEM_MOVEABLE As Long = &H2
Private Const CF_UNICODETEXT As Long = 13


'===============================
' 複数選択したメールの差出人アドレスを
' 重複なしで改行区切りコピー
'===============================
' 一覧で選択したメールの差出人アドレスを自動的にクリップボードに入れる
' ショートカット：Ctrl＋Shift＋C
' 2025/10/15
' 備考：
'   複数の場合、重複なしで改行区切りコピー
'===============================
Public Sub CopySenderAddress()
    On Error GoTo EH
    Dim sel As Selection
    Dim i As Long
    Dim itm As Object
    Dim dict As Object ' Scripting.Dictionary
    Dim addr As String
    Dim cnt As Long
    
    Set sel = Application.ActiveExplorer.Selection
    If sel Is Nothing Or sel.Count = 0 Then Exit Sub
    
    Set dict = CreateObject("Scripting.Dictionary")
    dict.CompareMode = 1 ' TextCompare
    
    For i = 1 To sel.Count
        Set itm = sel.Item(i)
        If TypeOf itm Is MailItem Then
            addr = GetSenderSmtpAddress(itm)
            If Len(addr) > 0 Then
                If Not dict.Exists(addr) Then dict.Add addr, True
            End If
        End If
    Next i
    
    If dict.Count = 0 Then
        MsgBox "コピー対象の差出人アドレスが見つかりませんでした。", vbInformation
        Exit Sub
    End If
    
    Dim result As String
    result = JoinDictionaryKeys(dict, vbCrLf) ' 改行区切り
    
    PutTextToClipboard result
    MsgBox "差出人アドレス（" & dict.Count & "件）をコピーしました。", vbInformation
    Exit Sub
EH:
    MsgBox "エラー: " & Err.Number & vbCrLf & Err.Description, vbExclamation
End Sub

'--- 選択した MailItem から SMTP アドレスを推定して取得
Private Function GetSenderSmtpAddress(mi As MailItem) As String
    On Error Resume Next
    
    ' 1) まず標準プロパティ
    Dim t As String
    t = mi.SenderEmailType ' "SMTP" or "EX"
    If UCase$(t) = "SMTP" Then
        GetSenderSmtpAddress = mi.SenderEmailAddress
        If Len(GetSenderSmtpAddress) > 0 Then Exit Function
    End If
    
    ' 2) Exchangeユーザーの場合（EX）
    If Not mi.Sender Is Nothing Then
        If UCase$(t) = "EX" Then
            Dim exu As Outlook.ExchangeUser
            Set exu = mi.Sender.GetExchangeUser
            If Not exu Is Nothing Then
                GetSenderSmtpAddress = exu.PrimarySmtpAddress
                If Len(GetSenderSmtpAddress) > 0 Then Exit Function
            End If
        End If
    End If
    
    ' 3) MAPI プロパティからの取得（PR_SMTP_ADDRESS）
    Dim pa As PropertyAccessor
    Set pa = mi.PropertyAccessor
    Const PR_SMTP_ADDRESS As String = "http://schemas.microsoft.com/mapi/proptag/0x39FE001E"
    Dim smtp As String
    smtp = pa.GetProperty(PR_SMTP_ADDRESS)
    If Len(smtp) > 0 Then
        GetSenderSmtpAddress = smtp
        Exit Function
    End If
    
    ' 4) それでも取れない場合のフォールバック（表示名を返す）
    GetSenderSmtpAddress = mi.SenderEmailAddress
End Function

'--- Dictionary のキーを区切り文字で結合
Private Function JoinDictionaryKeys(ByVal d As Object, ByVal sep As String) As String
    Dim keys() As Variant
    keys = d.keys
    JoinDictionaryKeys = Join(keys, sep)
End Function

'--- 文字列をクリップボードへ（MSForms.DataObject 不使用版）
Private Sub PutTextToClipboard(ByVal s As String)
    On Error GoTo EH
    Dim cbBytes As LongPtr
    Dim hMem As LongPtr
    Dim pMem As LongPtr

    ' 末尾のヌル文字分も確保（Unicodeなので×2）
    cbBytes = (Len(s) + 1) * 2
    hMem = GlobalAlloc(GMEM_MOVEABLE, cbBytes)
    If hMem = 0 Then Exit Sub

    pMem = GlobalLock(hMem)
    If pMem = 0 Then GoTo Cleanup

    ' 文字列をメモリへコピー
    lstrcpyW pMem, StrPtr(s)
    Call GlobalUnlock(hMem)

    If OpenClipboard(0) <> 0 Then
        Call EmptyClipboard
        ' SetClipboardDataが成功するとhMemの所有権はOSへ移る
        If SetClipboardData(CF_UNICODETEXT, hMem) <> 0 Then
            hMem = 0
        End If
        Call CloseClipboard
    End If

Cleanup:
    If hMem <> 0 Then
        ' 失敗時のみこちらで解放（成功時はOSが保持）
        Call GlobalUnlock(hMem)
    End If
    Exit Sub
EH:
    ' 必要ならエラー表示
    ' MsgBox "Clipboard error: " & Err.Number & " " & Err.Description, vbExclamation
End Sub
